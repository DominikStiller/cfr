Load test data to ClimateDataset#

[2]:

dirpath = '/Volumes/ExtraDrive/Data/GCM_sims/icesm_last_millennium_historical'

ds_tas = cfr.ClimateDataset().load_nc(os.path.join(dirpath, 'tas_sfc_Amon_iCESM_past1000historical_085001-200512.nc'))
